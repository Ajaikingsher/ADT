#include <iostream>
#include <vector>
using namespace std;

void sieveOfSundaram(int n) {
    int nNew = (n - 1) / 2;
    vector<bool> marked(nNew + 1, false);

    for (int i = 1; i <= nNew; i++)
        for (int j = i; i + j + 2 * i * j <= nNew; j++)
            marked[i + j + 2 * i * j] = true;

    if (n > 2) cout << 2 << " ";
    for (int i = 1; i <= nNew; i++)
        if (!marked[i]) cout << 2 * i + 1 << " ";
}

int main() {
    int n = 20;
    sieveOfSundaram(n);
    return 0;
}

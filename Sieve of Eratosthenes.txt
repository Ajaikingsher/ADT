#include <iostream>
#include <vector>
using namespace std;

vector<int> sieve(int n) {
    vector<bool> prime(n + 1, true);
    vector<int> res;

    for (int p = 2; p <= n; p++) {
        if (prime[p]) {
            res.push_back(p);
            for (int i = p * p; i <= n; i += p)
                prime[i] = false;
        }
    }
    return res;
}

int main() {
    int n ;cin>>n;
    for (int p : sieve(n))
        cout << p << " ";
    return 0;
}
